# Consumed by incept-first-run-green.sh. STACK-AGNOSTIC engine + per-archetype
# data: every TS/Node-specific token lives HERE, never in the check body. KW4/KW8
# add one manifest per stack (same grammar) with zero check rework.
# Security gates: assert the ENFORCING VALUE (a threshold/deny-flag/scope), not mere presence — a
# present-but-inert gate greens a presence-only assert (see DEVELOPMENT-STANDARDS.md §14; the sast-scoped
# block's `absent --config auto` + `--error .` scope is the pattern to copy).
#
# Block grammar (blank-line-separated; `key: value` lines; unknown keys ignored):
#   id:            <mechanism id>                  (required; names the mechanism)
#   precondition:  <predicate>                     (when is the mechanism EXPECTED live?)
#   when-absent:   na | skip                       (precondition false -> omit vs skip-with-reason)
#   assert:        <predicate>                      (config-shape; repeatable; ALL must hold)
#   skip-reason:   <plain text>                     (printed on SKIP/N/A)
#   fix-hint:      <plain text>                     (printed on MISCONFIGURED-RED)
#
# Predicate grammar (shared by precondition: and assert:):
#   present <ERE>    LIVE ci.yml (comments stripped) has a line matching <ERE>   (grep -E)
#   absent  <ERE>    LIVE ci.yml (comments stripped) has NO line matching <ERE>
#   comment <ERE>    RAW ci.yml (comments included) matches — for intentional in-repo
#                    explanations that are by-design a comment (e.g. a legible skip-reason)
#   scaffold <ERE>   some file under scaffold/ matches <ERE>                     (grep -rE)
#   scaffoldfile <relpath> <ERE>  scaffold/<relpath> exists AND matches <ERE>    (file-scoped grep -E)
#   file <relpath>   <archetype>/<relpath> exists
#   always           always true
# NOTE: present/absent read the comment-stripped view so a token that lives only in a comment
# (e.g. "never --config auto") neither satisfies present nor trips absent — assertions bind LIVE config.
#
# Resolution per mechanism:
#   precondition holds  + all asserts hold  -> GREEN
#   precondition holds  + an assert fails   -> MISCONFIGURED-RED (exit non-zero; names the mechanism)
#   precondition false  + when-absent=skip  -> SKIPPED-WITH-REASON
#   precondition false  + when-absent=na    -> N/A (omitted; genuinely not applicable)
# ---------------------------------------------------------------------------

id: sast-scoped
precondition: present semgrep
when-absent: skip
# EXCLUSION-scoping (owner-ratified): scan the whole app by default (`--error .`) so a new code root
# (app/, server/, lib/, pages/) stays covered without re-editing CI; opt OUT of only generated/vendored
# noise via a scaffold .semgrepignore. The whole-tree assertion below FAILS a NARROWED single-subdir
# scope (e.g. `--error src`) — a narrowed gate is a silent coverage loss, so it must RED, not green.
assert: present --config p/default
assert: absent --config auto
assert: present semgrep.*--error[[:space:]]+\.([[:space:]]|$)
assert: file scaffold/.semgrepignore
skip-reason: no SAST step in this profile — nothing to scope
fix-hint: scan the whole app (`semgrep --config p/default --error .`) so new code roots stay covered; ship a scaffold/.semgrepignore opting OUT of only generated/vendored noise (dist/, coverage/, node_modules/); keep p/default, never --config auto; do NOT inclusion-scope to one subdir
